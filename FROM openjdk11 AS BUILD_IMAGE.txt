FROM openjdk:11 AS BUILD_IMAGE
WORKDIR /usr/src/app
RUN apt update && apt install maven -y
COPY ../../* /usr/src/app/
RUN mvn install 

FROM tomcat:9-jre11

RUN rm -rf /usr/local/tomcat/webapps/*

COPY --from=BUILD_IMAGE /usr/src/app//target/onlinebookstore.war /usr/local/tomcat/webapps/ROOT.war

EXPOSE 8080
CMD ["catalina.sh", "run"]
